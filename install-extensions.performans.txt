# PowerShell Script — “extensions.performance.txt içindeki tüm VSCode eklentilerini yükle”

# Not: VSCode kapalı olsun → PowerShell içinde bu script’i çalıştır.
-------------------------------

# Path to performance extension list
$extList = "extensions.performance.txt"

# Load extension IDs (trim + boş satırları çıkar)
$extensions = Get-Content $extList | ForEach-Object { $_.Trim() } | Where-Object { $_ -ne "" }

Write-Host "`nInstalling extensions from extensions.performance.txt..." -ForegroundColor Cyan

foreach ($ext in $extensions) {
    Write-Host "Installing $ext ..." -ForegroundColor Yellow
    code --install-extension $ext --force
}

Write-Host "`nAll extensions installed successfully." -ForegroundColor Green
